    <script>
        // Dados Persistentes
        let contents = JSON.parse(localStorage.getItem('vibeflix_pro_db')) || [];

        function renderMovies(data = contents) {
            const grid = document.getElementById('movieGrid');
            if (!grid) return;
            
            grid.innerHTML = data.length ? '' : `
                <div class="col-span-full py-20 text-center">
                    <i class="fas fa-film text-slate-800 text-5xl mb-4"></i>
                    <p class="text-slate-600 font-bold">A tua biblioteca está vazia.</p>
                    <p class="text-slate-700 text-xs">Clica em "Novo" para adicionar o teu primeiro anime ou filme.</p>
                </div>
            `;
            
            data.forEach(item => {
                grid.innerHTML += `
                    <div class="movie-card group cursor-pointer" onclick="setRoute(${item.id})">
                        <div class="aspect-[2/3] rounded-[2rem] overflow-hidden relative shadow-2xl border border-white/5">
                            <img src="${item.cover}" class="w-full h-full object-cover transition duration-700 group-hover:scale-110" onerror="this.src='https://via.placeholder.com/500x750?text=Capa+Indisponivel'">
                            <div class="absolute inset-0 bg-gradient-to-t from-black via-black/20 to-transparent"></div>
                            
                            <div class="absolute bottom-5 left-5 right-5">
                                <span class="bg-cyan-500/20 text-cyan-400 text-[9px] font-black px-2 py-0.5 rounded uppercase mb-2 inline-block border border-cyan-500/30">${item.category}</span>
                                <h4 class="font-bold text-sm leading-tight group-hover:text-cyan-400 transition-colors">${item.title}</h4>
                            </div>
                            
                            <button onclick="event.stopPropagation(); deleteItem(${item.id})" class="absolute top-4 right-4 w-8 h-8 rounded-full bg-red-600/10 hover:bg-red-600 text-white opacity-0 group-hover:opacity-100 transition flex items-center justify-center backdrop-blur-md">
                                <i class="fas fa-trash text-[10px]"></i>
                            </button>
                        </div>
                    </div>
                `;
            });
        }

        // Sistema de Navegação Interna para o RAVE
        function setRoute(id) {
            window.location.hash = `watch/${id}`;
            openPlayer(id);
        }

        function openPlayer(id) {
            const item = contents.find(c => c.id == id);
            if (!item) return;

            // Atualiza tags dinamicamente (para o robô do RAVE)
            document.getElementById('metaTitle').content = item.title;
            document.getElementById('metaImage').content = item.cover;

            const player = document.getElementById('playerContent');
            player.innerHTML = `
                <div class="lg:w-2/5 relative h-[300px] lg:h-auto overflow-hidden">
                    <img src="${item.cover}" class="w-full h-full object-cover">
                    <div class="absolute inset-0 bg-gradient-to-r from-slate-950 via-transparent to-transparent hidden lg:block"></div>
                </div>
                <div class="lg:w-3/5 p-8 lg:p-16 flex flex-col justify-center bg-slate-950">
                    <button onclick="closePlayer()" class="absolute top-8 right-8 text-slate-500 hover:text-white transition"><i class="fas fa-times text-2xl"></i></button>
                    
                    <div class="mb-4 flex items-center gap-3">
                        <span class="text-cyan-400 font-black uppercase text-[10px] tracking-widest bg-cyan-400/10 px-3 py-1 rounded-full border border-cyan-400/20">${item.category}</span>
                        <span class="text-slate-500 text-xs font-bold">${item.year}</span>
                    </div>
                    
                    <h2 class="text-4xl lg:text-5xl font-black mb-6 italic brand-gradient leading-tight">${item.title}</h2>
                    
                    <div class="bg-white/5 p-6 rounded-[2rem] border border-white/5 mb-8">
                        <p class="text-slate-400 text-sm leading-relaxed">
                            O conteúdo está pronto. Se estiveres no <b>App Rave</b>, clica no botão abaixo para iniciar a sincronização.
                        </p>
                    </div>

                    <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                        <a href="${item.raveLink}" class="action-btn text-white py-5 rounded-2xl font-black text-center shadow-xl shadow-cyan-500/20 flex items-center justify-center gap-3">
                            <i class="fas fa-play"></i> ABRIR NO RAVE
                        </a>
                        <button onclick="copyLink('${item.raveLink}')" class="bg-slate-900 text-white py-5 rounded-2xl font-bold border border-white/5 hover:bg-slate-800 transition">
                            COPIAR LINK
                        </button>
                    </div>
                </div>
            `;
            document.getElementById('playerModal').style.display = 'flex';
            document.body.style.overflow = 'hidden';
        }

        function closePlayer() {
            window.location.hash = '';
            document.getElementById('playerModal').style.display = 'none';
            document.body.style.overflow = 'auto';
        }

        // Gestão de Dados
        function handleUpload(e) {
            e.preventDefault();
            const newItem = {
                id: Date.now(),
                title: document.getElementById('in_title').value,
                category: document.getElementById('in_cat').value,
                year: document.getElementById('in_year').value || "2026",
                cover: document.getElementById('in_cover').value,
                raveLink: document.getElementById('in_link').value
            };
            contents.unshift(newItem);
            localStorage.setItem('vibeflix_pro_db', JSON.stringify(contents));
            renderMovies();
            toggleModal('uploadModal');
            e.target.reset();
        }

        function deleteItem(id) {
            if(!confirm("Desejas remover este item?")) return;
            contents = contents.filter(c => c.id !== id);
            localStorage.setItem('vibeflix_pro_db', JSON.stringify(contents));
            renderMovies();
        }

        function searchMovie() {
            const query = document.getElementById('searchBox').value.toLowerCase();
            const filtered = contents.filter(i => i.title.toLowerCase().includes(query));
            renderMovies(filtered);
        }

        function filterContent(cat) {
            const title = document.getElementById('sectionTitle');
            if (cat === 'all') {
                title.innerText = "A Minha Biblioteca";
                renderMovies(contents);
            } else {
                title.innerText = cat.charAt(0).toUpperCase() + cat.slice(1) + "s";
                renderMovies(contents.filter(i => i.category === cat));
            }
        }

        function toggleModal(id) {
            const m = document.getElementById(id);
            m.classList.toggle('hidden');
        }

        function copyLink(url) {
            const el = document.createElement('textarea');
            el.value = url;
            document.body.appendChild(el);
            el.select();
            document.execCommand('copy');
            document.body.removeChild(el);
            alert("Link copiado para a área de transferência!");
        }

        // Backup & Restore
        function saveData() {
            const data = JSON.stringify(contents);
            const blob = new Blob([data], {type: 'application/json'});
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'vibeflix_backup.json';
            a.click();
            URL.revokeObjectURL(url);
        }

        function loadData(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (event) => {
                try {
                    contents = JSON.parse(event.target.result);
                    localStorage.setItem('vibeflix_pro_db', JSON.stringify(contents));
                    renderMovies();
                    alert("Dados importados com sucesso!");
                } catch(err) { alert("Erro ao importar ficheiro!"); }
            };
            reader.readAsText(file);
        }

        // Monitor de Rota (Hash)
        window.onhashchange = () => {
            const hash = window.location.hash;
            if (!hash) closePlayer();
        };

        window.onload = () => {
            renderMovies();
            const hash = window.location.hash;
            if (hash.startsWith('#watch/')) {
                const id = hash.split('/')[1];
                openPlayer(id);
            }
        };
    </script>
